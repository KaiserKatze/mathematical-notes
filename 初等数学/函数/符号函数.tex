\section{符号函数}
\begin{definition}[符号函数]
函数\[
	f(x) = \left\{ \begin{array}{cc}
		1, & x > 0 \\
		0, & x = 0 \\
		-1, & x < 0 \\
	\end{array} \right.
\]称为\DefineConcept{符号函数}，
记作\(\sgn x\).
\end{definition}
