\section{几种紧致性以及其间的关系}
\begin{definition}
%@see: 《点集拓扑讲义（第四版）》（熊金城） P214 定义7.4.1
设\(X\)是一个拓扑空间.
如果\(X\)的每一个可数开覆盖都有有限子覆盖，
则称“拓扑空间\(X\)是\DefineConcept{可数紧致的}”
“拓扑空间\(X\)具有\DefineConcept{可数紧致性}”
或“\(X\)是一个\DefineConcept{可数紧致空间}”.
\end{definition}

\begin{theorem}
%@see: 《点集拓扑讲义（第四版）》（熊金城） P214 定理7.4.1
任意一个紧致空间都是一个可数紧致空间.

\end{theorem}

\begin{definition}
%@see: 《点集拓扑讲义（第四版）》（熊金城） P214 定义7.4.2
设\(X\)是一个拓扑空间.
如果\(X\)的每一个无限子集都有聚点，
则称“拓扑空间\(X\)是\DefineConcept{列紧的}”
“拓扑空间\(X\)具有\DefineConcept{列紧性}”
或“\(X\)是一个\DefineConcept{列紧空间}”.
\end{definition}

%@see: 《点集拓扑讲义（第四版）》（熊金城） P214 定理7.4.3

\begin{definition}
%@see: 《点集拓扑讲义（第四版）》（熊金城） P215 定义7.4.3
设\(X\)是一个拓扑空间，
\(\{A_n\}_{n\in\mathbb{Z}^+}\)是\(X\)中的一个单调减序列，
且\(A_n\ (n=1,2,\dotsc)\)都是非空闭集，
则称“\(\{A_n\}_{n\in\mathbb{Z}^+}\)是\(X\)中的一个\DefineConcept{非空闭集单调减序列}”.
\end{definition}

\begin{lemma}
%@see: 《点集拓扑讲义（第四版）》（熊金城） P215 引理7.4.4
设\(X\)是一个拓扑空间，
则\(X\)是一个可数紧致空间，
当且仅当\(X\)中任意一个非空闭集单调减序列\(\{A_n\}_{n\in\mathbb{Z}^+}\)有非空的交，
即\(\bigcap_{n\in\mathbb{Z}^+} A_n \neq \emptyset\).
%TODO proof
\end{lemma}

\begin{definition}
%@see: 《点集拓扑讲义（第四版）》（熊金城） P216 定义7.4.4
设\(X\)是一个拓扑空间.
如果\(X\)中每一个序列都有一个收敛的子序列，
则称“拓扑空间\(X\)是\DefineConcept{序列紧致的}”
“拓扑空间\(X\)具有\DefineConcept{序列紧致性}”
或“\(X\)是一个\DefineConcept{序列紧致空间}”.
\end{definition}

%@see: 《点集拓扑讲义（第四版）》（熊金城） P216 定理7.4.6

%@see: 《点集拓扑讲义（第四版）》（熊金城） P217 定理7.4.7
任意一个满足第一可数性公理的可数紧致空间都是序列紧致空间.
%TODO proof
\end{theorem}

\begin{corollary}
%@see: 《点集拓扑讲义（第四版）》（熊金城） P217 推论7.4.8
设\(X\)是一个满足第二可数性公理的\(T_1\)空间，
\(A\)是\(X\)的一个子集，
则以下命题等价：\begin{itemize}
	\item \(A\)的每一个开覆盖都有有限子覆盖；
	\item \(A\)的每一个可数开覆盖都有有限子覆盖；
	\item \(A\)中每一个序列都有子序列收敛于\(A\)中的点；
	\item \(A\)中每一个无限子集都有聚点在\(A\)中.
\end{itemize}
%TODO proof
\end{corollary}

证明以下命题等价：\begin{itemize}
	\item \(A\)的每一个开覆盖都有有限子覆盖；
	\item \(A\)的每一个可数开覆盖都有有限子覆盖；
	\item \(A\)中每一个序列都有子序列收敛于\(A\)中的点；
	\item \(A\)中每一个无限子集都有聚点在\(A\)中；
	\item \(A\)是一个有界闭集.
\end{itemize}
