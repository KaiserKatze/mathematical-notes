\section{函数矩阵对纯量的导数与积分}
\subsection{函数矩阵的基本运算}
%@see: 《矩阵分析（第3版）》（史荣昌、魏丰） P229
以实变量\(x\)的实值函数\(a_{ij}(x)\ (i=1,2,\dotsc,m;j=1,2,\dotsc,n)\)为元素的矩阵\begin{equation*}
	A(x)
	\defeq
	\begin{bmatrix}
		a_{11}(x) & a_{12}(x) & \dots & a_{1n}(x) \\
		a_{21}(x) & a_{22}(x) & \dots & a_{2n}(x) \\
		\vdots & \vdots & & \vdots \\
		a_{m1}(x) & a_{m2}(x) & \dots & a_{mn}(x) \\
	\end{bmatrix}
\end{equation*}
称为\DefineConcept{函数矩阵}.

当行数\(m=1\)时，\(A(x)\)称为\DefineConcept{函数列向量}.
当列数\(n=1\)时，\(A(x)\)称为\DefineConcept{函数行向量}.

函数矩阵的加法、数量乘法，两个函数矩阵的乘法，函数矩阵的转置，
都与之前定义的常数矩阵的相应运算完全相同.
类似地，可以定义函数矩阵的伴随矩阵、逆矩阵.
\begin{definition}
%@see: 《矩阵分析（第3版）》（史荣昌、魏丰） P230 定义7.1.1
设\(A(x)\)是\(n\)阶函数矩阵.
若存在\(n\)阶函数矩阵\(B(x)\)
使得对于任何\(x \in [a,b]\)
都有\begin{equation*}
	A(x) B(x) = B(x) A(x) = E,
\end{equation*}
其中\(E\)是\(n\)阶单位矩阵，
则称
“\(A(x)\)是区间\([a,b]\)上的一个\DefineConcept{可逆（函数）矩阵}”
或称“\(A(x)\)在区间\([a,b]\)上\DefineConcept{可逆}”；
并称“\(B(x)\)是\(A(x)\)的\DefineConcept{逆（函数）矩阵}”，
记\(A^{-1}(x) \defeq B(x)\).
\end{definition}

\begin{theorem}
%@see: 《矩阵分析（第3版）》（史荣昌、魏丰） P230 定理7.1.1
\(n\)阶函数矩阵\(A(x)\)在区间\([a,b]\)上可逆的充分必要条件是
\(\DeterminantA{A(x)}\)在\([a,b]\)上处处不等于零，
并且\begin{equation*}
	A^{-1}(x)
	= \frac{1}{\DeterminantA{A(x)}}
	A^*(x),
\end{equation*}
其中\(A^*(x)\)是\(A(x)\)的伴随矩阵.
\end{theorem}

\subsection{函数矩阵的极限}
\begin{definition}
%@see: 《矩阵分析（第3版）》（史荣昌、魏丰） P231 定义7.1.2
如果函数矩阵\(A(x) = (a_{ij}(x))_{m \times n}\)的
每一个元素\(a_{ij}(x)\)都在点\(x_0\)收敛，
即\begin{equation*}
	\lim_{x \to x_0} a_{ij}(x)
	= a_{ij}
	\quad(i=1,2,\dotsc,m;j=1,2,\dotsc,n),
\end{equation*}
其中\(a_{ij}\)是与\(x\)无关的常数，
则称“\(A(x)\)在点\(x_0\)收敛”，
记为\(\lim_{x \to x_0} A(x) = A\)，
其中\begin{equation*}
	A \defeq \begin{bmatrix}
		a_{11} & a_{12} & \dots & a_{1n} \\
		a_{21} & a_{22} & \dots & a_{2n} \\
		\vdots & \vdots & & \vdots \\
		a_{m1} & a_{m2} & \dots & a_{mn} \\
	\end{bmatrix}.
\end{equation*}
\end{definition}

\begin{definition}
%@see: 《矩阵分析（第3版）》（史荣昌、魏丰） P231 定义7.1.2
如果函数矩阵\(A(x) = (a_{ij}(x))_{m \times n}\)的
每一个元素\(a_{ij}(x)\)都在\(x_0\)连续，
即\begin{equation*}
	\lim_{x \to x_0} a_{ij}(x)
	= a_{ij}(x_0)
	\quad(i=1,2,\dotsc,m;j=1,2,\dotsc,n),
\end{equation*}
则称“函数矩阵\(A(x)\)在点\(x_0\)连续”，
记为\(\lim_{x \to x_0} A(x) = A\)，
其中\begin{equation*}
	A \defeq \begin{bmatrix}
		a_{11} & a_{12} & \dots & a_{1n} \\
		a_{21} & a_{22} & \dots & a_{2n} \\
		\vdots & \vdots & & \vdots \\
		a_{m1} & a_{m2} & \dots & a_{mn} \\
	\end{bmatrix}.
\end{equation*}
\end{definition}

函数矩阵有着与函数相似的极限性质.
\begin{property}
%@see: 《矩阵分析（第3版）》（史荣昌、魏丰） P231
设\(
	\lim_{x \to x_0} A(x) = A \in M_{m \times t}(\mathbb{R}),
	\lim_{x \to x_0} B(x) = B \in M_{t \times n}(\mathbb{R})
\)，
则\begin{gather*}
	\lim_{x \to x_0} [ A(x) \pm B(x) ]
	= A \pm B, \\
	\lim_{x \to x_0} [ k A(x) ]
	= k A, \\
	\lim_{x \to x_0} [ A(x) B(x) ]
	= A B,
\end{gather*}
其中\(k\)是与\(x\)无关的常数.
\end{property}

\subsection{函数矩阵对纯量的导数}
\begin{definition}
%@see: 《矩阵分析（第3版）》（史荣昌、魏丰） P231 定义7.1.3
如果函数矩阵\(A(x) = (a_{ij}(x))_{m \times n}\)的
每一个元素\(a_{ij}(x)\)都在\(x_0\)可导，
则称“函数矩阵\(A(x)\)在点\(x_0\)可导”，
记\begin{equation*}
	A'(x_0)
	\equiv \eval{\dv{A(x)}{x}}_{x=x_0}
	\defeq \begin{bmatrix}
		a_{11}'(x_0) & a_{12}'(x_0) & \dots & a_{1n}'(x_0) \\
		a_{21}'(x_0) & a_{22}'(x_0) & \dots & a_{2n}'(x_0) \\
		\vdots & \vdots & & \vdots \\
		a_{m1}'(x_0) & a_{m2}'(x_0) & \dots & a_{mn}'(x_0) \\
	\end{bmatrix}.
\end{equation*}
\end{definition}

\begin{property}
%@see: 《矩阵分析（第3版）》（史荣昌、魏丰） P232
函数矩阵\(A(x)\)是常数矩阵的充分必要条件是\(A'(x)\)处处等于零.
\end{property}

\begin{property}
%@see: 《矩阵分析（第3版）》（史荣昌、魏丰） P232
设函数矩阵\(
	A(x) = (a_{ij}(x))_{m \times n},
	B(x) = (b_{ij}(x))_{m \times n}
\)都可导，
则\begin{equation*}
	\dv{x} [ A(x) + B(x) ]
	= \dv{A(x)}{x} + \dv{B(x)}{x}.
\end{equation*}
\end{property}

\begin{property}
%@see: 《矩阵分析（第3版）》（史荣昌、魏丰） P232
设\(k(x)\)是\(x\)的纯量函数，
\(A(x)\)是函数矩阵，
\(k(x)\)与\(A(x)\)都可导，
则\begin{equation*}
	\dv{x} [ k(x) A(x) ]
	= \dv{k(x)}{x} A(x)
	+ k(x) \dv{A(x)}{x}.
\end{equation*}
\end{property}

\begin{property}
%@see: 《矩阵分析（第3版）》（史荣昌、魏丰） P232
设\(
	A(x) = (a_{ij}(x))_{m \times t},
	B(x) = (b_{ij}(x))_{t \times n}
\)都可导，
则\begin{equation*}
	\dv{x} [ A(x) B(x) ]
	= \dv{A(x)}{x} B(x)
	+ A(x) \dv{B(x)}{x}.
\end{equation*}
\end{property}
\begin{remark}
因为矩阵乘法不服从交换律，
所以\begin{gather*}
	\dv{x} A^2(x)
	\neq 2 A(x) \dv{A(x)}{x}, \\
	\dv{x} A^3(x)
	\neq 3 A^2(x) \dv{A(x)}{x}.
\end{gather*}
\end{remark}

\begin{property}
%@see: 《矩阵分析（第3版）》（史荣昌、魏丰） P232
设\(A(x) = (a_{ij}(x))_n\)及其逆矩阵\(A^{-1}(x)\)都可导，
则\begin{equation*}
	\dv{A^{-1}(x)}{x}
	= - A^{-1}(x)
	\dv{A(x)}{x}
	A^{-1}(x).
\end{equation*}
\begin{proof}
因为\(A^{-1}(x) A(x) = E\)，
所以\begin{equation*}
	\dv{x} [ A^{-1}(x) A(x) ]
	= \dv{A^{-1}(x)}{x} A(x)
	+ A^{-1}(x) \dv{A(x)}{x}
	= \dv{x} E
	= 0,
\end{equation*}
于是\begin{equation*}
	\dv{A^{-1}(x)}{x}
	= - A^{-1}(x)
	\dv{A(x)}{x}
	A^{-1}(x).
	\qedhere
\end{equation*}
\end{proof}
\end{property}

\begin{property}
%@see: 《矩阵分析（第3版）》（史荣昌、魏丰） P233
设\(A(x)\)是函数矩阵，\(x = f(t)\)是\(t\)的纯量函数，
\(A(x)\)与\(f(t)\)都可导，
则\begin{equation*}
	\dv{A(x(t))}{t}
	= \dv{A(x)}{x} f'(t).
\end{equation*}
\end{property}

函数矩阵的导数也是一个函数矩阵，
只要可导就能继续求导，得到函数矩阵的高阶导数：
\begin{align*}
	\dv[2]{A(x)}{x}
	&\defeq \dv{x}\left( \dv{A(x)}{x} \right), \\
	\dv[3]{A(x)}{x}
	&\defeq \dv{x}\left( \dv[2]{A(x)}{x} \right), \\
	&\hspace{6pt}\vdots \\
	\dv[k+1]{A(x)}{x}
	&\defeq \dv{x}\left( \dv[k]{A(x)}{x} \right).
\end{align*}

\subsection{函数矩阵的积分}
\begin{definition}
%@see: 《矩阵分析（第3版）》（史荣昌、魏丰） P233 定义7.1.4
如果函数矩阵\(A(x) = (a_{ij}(x))_{m \times n}\)的每一个元素\(a_{ij}(x)\)
都在区间\([a,b]\)上可积，
则称“函数矩阵\(A(x)\)在\([a,b]\)上可积”，
记\begin{equation*}
	\int_a^b A(x) \dd{x}
	\defeq
	\begin{bmatrix}
		\int_a^b a_{11}(x_0) \dd{x} & \int_a^b a_{12}(x_0) \dd{x} & \dots & \int_a^b a_{1n}(x_0) \dd{x} \\
		\int_a^b a_{21}(x_0) \dd{x} & \int_a^b a_{22}(x_0) \dd{x} & \dots & \int_a^b a_{2n}(x_0) \dd{x} \\
		\vdots & \vdots & & \vdots \\
		\int_a^b a_{m1}(x_0) \dd{x} & \int_a^b a_{m2}(x_0) \dd{x} & \dots & \int_a^b a_{mn}(x_0) \dd{x} \\
	\end{bmatrix}.
\end{equation*}
\end{definition}

\begin{property}
%@see: 《矩阵分析（第3版）》（史荣昌、魏丰） P233
设函数矩阵\(A(x)\)在\([a,b]\)上可积，
\(k\)是与\(x\)无关的常数，
则\begin{equation*}
	\int_a^b k A(x) \dd{x}
	= k \int_a^b A(x) \dd{x}.
\end{equation*}
\end{property}

\begin{property}
%@see: 《矩阵分析（第3版）》（史荣昌、魏丰） P233
设函数矩阵\(
	A(x) = (a_{ij}(x))_{m \times n},
	B(x) = (b_{ij}(x))_{m \times n}
\)都在\([a,b]\)上可积，
则\begin{equation*}
	\int_a^b k [ A(x) + B(x) ] \dd{x}
	= \int_a^b A(x) \dd{x}
	+ \int_a^b B(x) \dd{x}.
\end{equation*}
\end{property}
